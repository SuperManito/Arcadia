import APITable from '@site/src/components/APITable';
import { GET } from '@site/src/components/RequestTag';

#### <GET>/config/type/:type</GET>

## 请求

<APITable>

| 名称 | 类型 | 必填 | 描述 |
| --- | :-: | :-: | --- |
| `type` | `string` | 是 | 类型 |

</APITable>

## 响应

<APITable>

| 名称 | 类型 | 描述 |
| --- | :-: | --- |
| `id` | `number` | id |
| `type` | `string` | 类型 |
| `label` | `string` | 名称 |
| `value` | `string` | 值 |

</APITable>

```json title="示例"
{
  "data": [
    {

      "sort": 0
    },

      "type": "",
      "label": "",
      "value": "",
      "sort": 0
    },
    ...
  ],
}
```
